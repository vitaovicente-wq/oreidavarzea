function atualizarResumo(time) { const forca = time.forca || 60; // 1. Tenta pegar do objeto do time (se tiver no DB) let estadio = time.estadio; // 2. Se não tiver, usa lista de estádios reais conhecidos if (!estadio) { const estadiosReais = { "Corinthians": "Neo Química Arena", "Palmeiras": "Allianz Parque", "São Paulo": "Morumbi", "Santos": "Vila Belmiro", "Flamengo": "Maracanã", "Fluminense": "Maracanã", "Vasco": "São Januário", "Botafogo": "Nilton Santos", "Grêmio": "Arena do Grêmio", "Internacional": "Beira-Rio", "Atlético-MG": "Arena MRV", "Cruzeiro": "Mineirão", "Bahia": "Fonte Nova", "Vitória": "Barradão", "Fortaleza": "Castelão", "Ceará": "Castelão", "Athletico-PR": "Ligga Arena", "Coritiba": "Couto Pereira" }; estadio = estadiosReais[time.nome] || null; } // 3. Fallback genérico se ainda não achou if (!estadio) { if (forca >= 80) estadio = "Arena Multiuso"; else if (forca >= 70) estadio = "Estádio Municipal"; else estadio = "Campo do Bairro"; } // Define Objetivo e Orçamento let objetivo = "Permanência"; let orcamento = 2.0; if (forca >= 82) { objetivo = "Título / Libertadores"; orcamento = 150.0; } else if (forca >= 76) { objetivo = "Vaga na Libertadores"; orcamento = 60.0; } else if (forca >= 72) { objetivo = "Vaga na Sul-Americana"; orcamento = 25.0; } // Craque let craque = "---"; if (time.elenco && time.elenco.length > 0) { const melhor = time.elenco.reduce((p, c) => (p.forca > c.forca) ? p : c); craque = melhor.nome; } // Atualiza na tela document.getElementById('resumo-obj').innerText = objetivo; document.getElementById('resumo-orcamento').innerText = `R$ ${orcamento} Mi`; document.getElementById('resumo-estadio').innerText = estadio; document.getElementById('resumo-craque').innerText = craque; }
