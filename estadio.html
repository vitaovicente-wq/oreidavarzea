function calcularProjecao() { // 1. Pega os valores da tela const precosTela = { geral: parseInt(document.getElementById('range-geral').value), cadeiras: parseInt(document.getElementById('range-cadeiras').value), vip: parseInt(document.getElementById('range-vip').value), estacionamento: parseInt(document.getElementById('range-park').value) // Adicionei Estacionamento }; // 2. Manda para a Engine temporariamente Engine.Estadios.salvarPrecos(precosTela); // 3. Simula contra um time "Padrão" (Força 70) const simulacao = Engine.Estadios.calcularBilheteria({ forca: 70 }); // 4. Limpa o temporário Engine.Estadios._tempPrecos = null; // 5. Atualiza a tela // Público document.getElementById('proj-pub').innerText = simulacao.publicoTotal.toLocaleString() + " pagantes"; // Ocupação const elOcup = document.getElementById('proj-ocup'); elOcup.innerText = simulacao.ocupacao + "%"; // Cores dinâmicas if(simulacao.ocupacao < 30) elOcup.style.color = "#ff4757"; // Ruim else if(simulacao.ocupacao > 80) elOcup.style.color = "#00ff88"; // Bom else elOcup.style.color = "#fff"; // Dinheiro document.getElementById('proj-money').innerText = simulacao.rendaTotal.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' }); }
